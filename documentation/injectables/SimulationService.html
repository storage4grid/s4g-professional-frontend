import { Injectable } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';

import { AppSettings } from '../app-config';

import { KeycloakService } from 'keycloak-angular';

import { GridService } from './grid.service';

// ******** Models *********
import { Duration, Simulation, SimulationResult } from '../models/index';

/**
* Simulation service for communicating with the DSF-SE and handling the simulation
*/
@Injectable()
export class SimulationService {

  /**
  * Base URL of the DSF-SE API
  */
  baseUrl: string;

  /**
  * Parameters for the simulation
  * @type Simulation
  */
  simulation: Simulation;

  /**
  * ID for the simulation
  */
  simulationId: any;

  /**
  * Radials selected for simulation
  */
  simulationRadials = [];

  /**
  * Shows if there is currently a simulation being processed
  * @type boolean
  */
  simulationRunning = false;

  constructor ( private http: HttpClient,
                private keycloak: KeycloakService,
                private gridService: GridService ) {
    this.reset();

    this.baseUrl = AppSettings.DSF_SE_API_URL;
  }

  /**
  * Check if a radial is selected
  */
  hasSelectedRadials() {
    if ( this.simulationRadials.length == 0 ) {
      return false;
    } else {
      return true;
    }
  }

  /**
  * Set a selected radial
  *
  * @param radialId The ID of the radial that has been selected
  */
  selectRadial( radialId ) {
    this.simulationRadials.push(radialId);
  }

  /**
  * Deselect a radial
  *
  * @param radialId The ID of the radial that has been selected
  */
  deselectRadial( radialId ) {
    let index = this.simulationRadials.indexOf(radialId, 0);

    if (index > -1) {
       this.simulationRadials.splice(index, 1);
    }
  }

  /**
  * Deselect all radials
  *
  */
  deselectAllRadials() {
    this.simulationRadials = [];
  }

  /**
  * Get the currently selected radial
  *
  * @returns The ID of the selected radial
  */
  getSelectedRadials() {
    return this.simulationRadials;
  }

  /**
  * Reset all simulation parameters
  */
  reset() {
    this.simulation = {
      gridId: '',
      duration: {
        day: 0,
        month: 0,
        year: 0
      },
      thresholdLow: 0,
      thresholdMedium: 0,
      thresholdHigh: 0
    };

    this.simulationRadials = [];

    this.simulationRunning = false;
  }

  /**
  * Start the simulation
  * @param duration The timeframe which will be simulated
  *
  * @returns Promise
  */
  runSimulation( duration: Duration ) {
    this.simulationRunning = true;
    this.simulation.duration = duration;

    return new Promise( (resolve, reject) => {
      // gridId is the radialId
      if ( this.hasSelectedRadials() ) {

        // get the simulation grid
       let radial = this.gridService.createGridForSimulation( this.simulationRadials );

       // set the threshold (currently dummy value)
       this.simulation.thresholdHigh = 0.5;
       this.simulation.thresholdMedium = 0.4;
       this.simulation.thresholdLow = 0.35;

       radial.radials[0].photovoltaics = [];

        // send the radial to the simulation engine
        this.http.post( this.baseUrl + 'simulation', radial )
          .toPromise()
          .then(
            (simulationId) => {
              this.simulationId = simulationId;

              // start the simulation
              this.http.post( this.baseUrl + 'commands/run/' + simulationId, this.simulation )
                .toPromise()
                .then(
                  (simulationResults) => {
                    this.simulationRunning = false;
                    resolve( simulationResults );
                  }
                )
                .catch(
                  (error) => {
                    this.simulationRunning = false;
                    console.log( error );
                    reject();
                  }
                );
            }
          )
          .catch(
            (error) => {
              this.simulationRunning = false;
              console.log( error );
              reject();
            }
          );

      } else {
        reject( 'No radial chosen for simulation' );
      }
    });
  }

  /**
  * Abort a running the simulation
  */
  abortSimulation() {
    this.simulationRunning = false;

    // TODO: tell the server
  }
}
