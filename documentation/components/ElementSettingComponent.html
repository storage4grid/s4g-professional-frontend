import { Component, OnInit, Inject } from '@angular/core';
import { MatDialogRef } from '@angular/material';
import {MAT_DIALOG_DATA} from '@angular/material';

@Component({
  selector: 'app-element-settings',
  templateUrl: './element-settings.component.html'
})

/**
* A component for the settings pop-up displayed when creating a new element on the grid
*/
export class ElementSettingComponent implements OnInit {

  elementType: any;

  constructor(public dialogRef: MatDialogRef<ElementSettingComponent>,
              @Inject(MAT_DIALOG_DATA) public data: any) {

    // set the element type
    this.elementType = data.elementType;
  }

  storageModels = [
    {
      id: 0,
      name : 'Fronius Solar Battery 12.0',
      manufacturer : 'Fronius',
      electricalParameters : {
        capacity : {
          usableCapacity : 9600,
          capacityEntity : 'Wh'
        },
        voltageRange : {
          voltageRangeMin : 320,
          voltageRangeMax : 460,
          voltageEntity : 'V'
        },
        cycleStability : 8000,
        charging : {
          nominalChargingPower : 6400,
          nominalChargingPowerEntity : 'W',
          maxChargingCurrent : 16,
          maxChargingEntity : 'A'
        },
        discharging : {
          nominalDischargingPower : 6400,
          nominalDischargingPowerEntity : 'W',
          maxDischargingCurrent : 16,
          maxChargingEntity : 'A'
        }
      }
    }
  ];

  selectedStorageModel;

  ngOnInit() {
  }

  onNoClick(): void {
    this.dialogRef.close();
  }

  cancel(): void {
    this.dialogRef.close();
  }

  createStorage(): void {
    this.dialogRef.close({
      storageType: this.storageModels[this.selectedStorageModel]
    });
  }

  createEV(): void {
    this.dialogRef.close({
      storageType: ''
    });
  }

  createPV(): void {
    this.dialogRef.close({
      storageType: ''
    });
  }

  createHouse(): void {
    this.dialogRef.close({
      storageType: ''
    });
  }
}

<!-- Settings for storage -->
<div *ngIf="elementType === 'storage'">
  <p>Please, choose a storage model</p>
  <mat-form-field>
    <mat-select placeholder="Storage model" name="selectedStorageModel" [(ngModel)]="selectedStorageModel">
      <mat-option *ngFor="let storageModel of storageModels; let i=index" [value]="i">
        {{ storageModel.name }}
      </mat-option>
    </mat-select>
  </mat-form-field>
  <div>
    <button mat-button (click)="cancel()">Cancel</button>
    <button mat-raised-button color="accent" (click)="createStorage()">Create storage</button>
  </div>
</div>
