import * as L from 'leaflet';
import 'leaflet.markercluster';

import { partition, uniqueId } from 'lodash';

// ******** Angular Modules *********
import { Component, OnInit, OnDestroy } from '@angular/core';
import { MatDialog, MatDialogConfig } from '@angular/material';
import { ActivatedRoute } from '@angular/router';

// ******** Components *********
import { PrivacyPromptComponent } from '../privacy-prompt/privacy-prompt.component';

import { AppSettings } from '../../app-config';
import { GridService } from '../../services/grid.service';

import { leafletOptions, markerClusterOptions } from './map.config';
import { GridElementsLayer, GridLayer } from './map.typings';

import { Nodes, Node, ConnectedElement } from '../../models/node';
import { Radials } from '../../models/radial';
import { House, Houses } from '../../models/house';
import { PV, PVs } from '../../models/pv';
import { Storage, Storages } from '../../models/storage';
import { Transformer, Transformers } from '../../models/transformer';
import { GridElements, Grid } from '../../models/grid';

import {
  createNodeMarkers,
  lineMarkers,
  createIconMarkers,
  extractConnectedElements,
  parseConnectedElement
} from './map.util';

/**
 * This component handles the map view, which is restricted to authenticated users only.
 * It also is the main component handling all actions involving element creation and simulation.
 */
@Component({
  selector: 'app-map',
  templateUrl: './map.component.html',
  styleUrls: ['map.component.scss']
})
export class MapComponent implements OnInit, OnDestroy {
  /**
   * Privacy prompt to inform the user about data issues
   */
  privacyPrompt: any;

  /**
   * open privacy prompt value, read from local storage
   **/

  showPrivacyPrompt: true;

  /* Search error */
  searchErrorMessage: string;

  /* Grid details */

  locations: string[] = Object.keys(AppSettings.LOCATIONS);

  status: string;

  /* Leaflet layers and other information */

  // Values to bind to Leaflet Directive

  map: L.Map;
  options: L.MapOptions = leafletOptions;

  location: string;

  gridElements: GridElements = {
    houses: {},
    pvs: {},
    storages: {},
    transformers: {}
  };
  gridElementsLayer: GridElementsLayer = {
    houseLayer: {},
    pvLayer: {},
    storageLayer: {},
    transformerLayer: {}
  };

  grid: Grid = {
    nodes: {},
    radials: {}
  };
  gridLayer: GridLayer = {
    nodeLayer: {},
    radialLayer: []
  };

  // Marker cluster stuff
  markerClusterGroup: L.MarkerClusterGroup;
  markerClusterData: L.Marker[] = [];
  markerClusterOptions: L.MarkerClusterGroupOptions = markerClusterOptions;

  constructor(
    private gridService: GridService,
    private route: ActivatedRoute,
    public dialog: MatDialog
  ) {
    if (!localStorage.getItem('defaultLocation')) {
      localStorage.setItem('defaultLocation', 'fur');
    }

    this.status = 'normal';
  }

  ngOnInit() {
    this.route.queryParams.subscribe(queryParams => {
      this.location =
        queryParams['location'] || localStorage.getItem('defaultLocation');
      this.populateData(this.location);

      if (this.map) {
        this.moveMapTo(this.location);
      }
    });

    // open privacyPrompt
    /**
     * open the privacy prompt dialog
     * needed to use setTimeout to prevent ExpressionChangedAfterItHasBeenCheckedError #5268; more info can be found
     * on the gitHub page https://github.com/angular/material2/issues/5268
     **/
    setTimeout(() => this.openPrivacyPrompt());
  }

  partitionConnectedElementsInNode(nodes: Nodes): [Houses, Storages, PVs] {
    let houses: Houses = {};
    let storages: Storages = {};
    let pvs: PVs = {};

    const connectedElements: ConnectedElement[] = Object.entries(nodes).map(
      node => parseConnectedElement(node)
    );

    connectedElements.forEach(connectedElement => {
      const [house, storage, pv] = this.partitionConnectedElement(
        connectedElement
      );

      houses = { ...houses, ...house };
      storages = { ...storages, ...storage };
      pvs = { ...pvs, ...pv };
    });

    return [houses, storages, pvs];
  }

  partitionConnectedElement(
    connectedElement: ConnectedElement
  ): [Houses, Storages, PVs] {
    let houses: Houses = {};
    let storages: Storages = {};
    let pvs: PVs = {};

    const noOfElements = connectedElement.elements.length;

    const nodeLatLng = connectedElement.coordinate;
    const nodePoint = this.map.latLngToContainerPoint(nodeLatLng);

    if (noOfElements) {
      const angle = Math.ceil(360 / noOfElements) % 360;
      const radius = Math.ceil(noOfElements / 6) * 25;

      const elementsWithCoordinate = connectedElement.elements.map(
        (element, index) => {
          const elementAngleInRads = (angle * index + 30) * (Math.PI / 180);

          const latitudine =
            nodePoint.x + Math.sin(elementAngleInRads) * radius;
          const longitudine =
            nodePoint.y + Math.cos(elementAngleInRads) * radius;

          const { lat, lng } = this.map.containerPointToLatLng(
            L.point(latitudine, longitudine, true)
          );

          return { latitudine: lat, longitudine: lng, ...element };
        }
      );

      const [houseOrStorage, pvPartition] = partition(
        elementsWithCoordinate,
        (element: House | Storage | PV) => 'category' in element
      );

      const [storagePartition, housePartition] = partition(
        houseOrStorage,
        (element: House | Storage) => element.category === 'storage'
      );

      houses = extractConnectedElements(housePartition);
      storages = extractConnectedElements(storagePartition);
      pvs = extractConnectedElements(pvPartition);
    }

    return [houses, storages, pvs];
  }

  populateData(location: string): void {
    this.gridService
      .getGridData(location)
      .subscribe(([nodes, radials, _, transformers]) => {
        this.grid = {
          nodes: nodes,
          radials: radials
        };

        const [houses, storages, pvs] = this.partitionConnectedElementsInNode(
          nodes
        );

        this.gridElements = {
          houses: houses,
          pvs: pvs,
          storages: storages,
          transformers: transformers
        };

        this.gridLayer.nodeLayer = createNodeMarkers(nodes);

        this.gridElementsLayer.houseLayer = createIconMarkers(
          houses,
          'house',
          'House'
        );

        this.gridElementsLayer.pvLayer = createIconMarkers(pvs, 'pv', 'PV');

        this.gridElementsLayer.transformerLayer = createIconMarkers(
          transformers,
          'transformer',
          'Transformer'
        );

        this.gridLayer.radialLayer = lineMarkers(
          this.gridElements,
          nodes,
          radials
        );

        this.displayData();
      });
  }

  displayData() {
    this.markerClusterGroup.addLayers(this.gridLayer.radialLayer);

    this.markerClusterGroup.addLayers(Object.values(this.gridLayer.nodeLayer));

    this.markerClusterGroup.addLayers(
      Object.values(this.gridElementsLayer.houseLayer)
    );
    this.markerClusterGroup.addLayers(
      Object.values(this.gridElementsLayer.pvLayer)
    );
    this.markerClusterGroup.addLayers(
      Object.values(this.gridElementsLayer.storageLayer)
    );
    this.markerClusterGroup.addLayers(
      Object.values(this.gridElementsLayer.transformerLayer)
    );
  }

  /* PROMPT */

  openPrivacyPrompt(): void {
    // check browserStorage if prompt should be displayed
    try {
      this.showPrivacyPrompt = JSON.parse(
        localStorage.getItem('S4GShowPrivacyPromptProfGui')
      );
    } catch (e) {
      this.showPrivacyPrompt = true;
    }
    if (this.showPrivacyPrompt || this.showPrivacyPrompt == null) {
      const dialogConfig = new MatDialogConfig();

      dialogConfig.disableClose = true;
      dialogConfig.autoFocus = true;
      dialogConfig.maxWidth = '600px';
      dialogConfig.maxHeight = '600px';

      this.privacyPrompt = this.dialog.open(
        PrivacyPromptComponent,
        dialogConfig
      );
    }
  }

  /* EVENTS */

  onMapReady(map: L.Map) {
    this.map = map;
  }

  markerClusterReady(group: L.MarkerClusterGroup) {
    this.markerClusterGroup = group;
  }

  moveMapTo(location: string) {
    const locations = AppSettings.LOCATIONS;

    if (location in locations) {
      this.map.panTo(locations[location]).setZoom(leafletOptions.zoom);
    } else {
      this.searchErrorMessage = 'Search location not found';
    }
  }

  toolSelection(activeTool: string) {
    if (activeTool === 'select') {
      Object.entries(this.gridLayer.nodeLayer).forEach(([tech_id, node]) =>
        node.setRadius(6).setStyle({
          fillColor: '#000'
        })
      );
    }

    if (activeTool === 'storage') {
      Object.entries(this.gridLayer.nodeLayer).forEach(([tech_id, node]) =>
        node
          .setRadius(8)
          .setStyle({
            fillColor: '#2B79D3'
          })
          .unbindTooltip()
          .on('click', event => {
            const storage_tech_id = uniqueId('storage_');

            const newStorage: Storage = {
              tech_id: storage_tech_id,
              phases: 1,
              bus1: tech_id,
              feeder: 'feeder_001',
              kV: 0.23,
              kW: 3,
              pf: 0.95,
              conn: 'wye',
              model: 1,
              category: 'storage',
              monitor: { state: 'ON', var: ['kW', 'kVA', 'voltage'] }
            };

            let clickedNode = this.grid.nodes[tech_id];

            clickedNode.connected_elements = {
              ...clickedNode.connected_elements,
              [storage_tech_id]: newStorage
            };

            const connectedElement: ConnectedElement = parseConnectedElement([
              clickedNode.tech_id,
              clickedNode
            ]);

            const [house, storage, pv] = this.partitionConnectedElement(
              connectedElement
            );

            this.gridElements.houses = {
              ...this.gridElements.houses,
              ...house
            };

            this.gridElements.storages = {
              ...this.gridElements.storages,
              ...storage
            };

            this.gridElements.pvs = { ...this.gridElements.pvs, ...pv };

            this.gridElementsLayer.houseLayer = {
              ...this.gridElementsLayer.houseLayer,
              ...createIconMarkers(house, 'house', 'House')
            };

            this.gridElementsLayer.storageLayer = {
              ...this.gridElementsLayer.storageLayer,
              ...createIconMarkers(storage, 'storage', 'Storage')
            };

            this.gridElementsLayer.pvLayer = {
              ...this.gridElementsLayer.pvLayer,
              ...createIconMarkers(pv, 'pv', 'PV')
            };

            const radial_tech_id = uniqueId('powerline_');

            const powerline = {
              tech_id: radial_tech_id,
              bus1: tech_id,
              bus2: storage_tech_id,
              linecode: 'CU_4x10',
              phases: 3,
              length: 0.022,
              monitor: {
                state: 'OFF',
                var: ['kW', 'kVA', 'voltage']
              }
            };

            this.grid.radials = {
              ...this.grid.radials,
              [radial_tech_id]: powerline
            };

            this.gridLayer.radialLayer = lineMarkers(
              this.gridElements,
              this.grid.nodes,
              this.grid.radials
            );

            this.markerClusterGroup.clearLayers();
            this.displayData();
            this.markerClusterGroup.refreshClusters();
          })
      );
    }
  }

  ngOnDestroy(): void {
    // Called once, before the instance is destroyed.
    // Add 'implements OnDestroy' to the class.
    this.map.off();
    this.map.remove();
  }
}
